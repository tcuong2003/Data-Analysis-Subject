a2 = a ** 2

v_min = np.min(a)
v_avg = np.mean(a)

print(f'\nd. Phuong sai: [{v_sigma2:.2f}] va do lech chuan: [{v_var:.2f}]')

import numpy as np

print("3. Xử lý khác trên dãy số")

# a) Tạo dãy số x có 10 phần tử từ 1 đến 10
x = np.arange(1, 11)  # np.arange(start, stop) → stop không bao gồm, nên dùng 11

print(f'a. Day so x: {x}')

# b) Liệt kê và đếm số lượng phần tử chẵn và lẻ trong a và x
achan = a[a % 2 == 0]
ale = a[a % 2 != 0]
xchan = x[x % 2 == 0]
xle = x[x % 2 != 0]

print(f'b. So luong a: Chan = {len(achan)}, Le = {len(ale)}')
print(f'   So luong x: Chan = {len(xchan)}, Le = {len(xle)}')

# c) Tính khoảng cách giữa các phần tử ở vị trí lẻ của mảng a và x
# i lẻ → chỉ số 0,2,4,... vì Python index từ 0
ai_le = a[1::2]  # các vị trí lẻ: 1,3,5,... index
xi_le = x[1::2]
s = np.sqrt(np.sum((ai_le - xi_le)**2))
print(f'c. Khoang cach s = {s:.2f}')

# d) Tìm khoảng cách nhỏ nhất giữa tập a và x
# dùng broadcasting hoặc meshgrid để tìm tất cả khoảng cách
distances = np.abs(a.reshape(-1, 1) - x.reshape(1, -1))
min_distance = np.min(distances)
print(f'd. Khoang cach nho nhat giua tap a va x: {min_distance}')

$$ y = f\left(x| \mu, \sigma\right) = \frac{1}{\sigma\sqrt{2\pi}}exp\left(-\frac{\left(x - \mu\right)^2}{2\sigma^2}\right) $$ nơi mà $x \in [\mu - 5\sigma, \mu + 5\sigma]$
(c) Sinh ngẫu nhiên dãy số có 100000 phần tử theo phân bố chuẩn $N(\mu = 1, \sigma^2 = 1.0)$ và vẽ đồ thị tần số xuất hiện

Giải hệ phương trình tuyến tính sau: $$ \left\{\begin{matrix} 4x_1 + 3x_2 - 5x_3 = 2 \\ -2x_1 - 4x_2 + 5x_3 = 5 \\ 8x_1 + 8x_2 = -3 \\ \end{matrix}\right. $$

import numpy as np

# Giả sử A là ma trận 4x4 và y là vector 4 phần tử
A = np.random.randint(1, 10, size=(4,4))
y = np.random.randint(1, 10, size=(4,))

# Giải hệ Ax = y
x = np.linalg.solve(A, y)  # x = A^-1 y

# Kiểm tra nghiệm
yy = A @ x
diff = np.linalg.norm(yy - y)  # chuẩn 2 của vector yy - y

print(f"+ Ma tran A: \n{A}")
print(f"+ Vector y: {y}")
print(f"+ Nghiem x: {x}")

Bài 3. Tính gần đúng số pi dùng phương pháp Monte Carlo¶

Vẽ một đường tròn $C$ có bán kính $r$ nội tiếp một hình vuông $C$, và đặt ngẫu nhiên $N$ chấm lên hình vuông. Tỉ lệ các chấm nằm trong hình tròn trên tổng số chấm xấp xỉ bằng diện tích của hình tròn chia cho hình vuông như sau: $$ \frac{N_C}{N} \approx \frac{\mathbb{S}_C}{\mathbb{S}_N} $$ nơi mà $N_C$ là các chấm trong hình tròn, $N$ là tổng số chấm, $\mathbb{S}_C$ và $\mathbb{S}_N$ lần lượt là diện tích của hình tròn và hình vuông. $$ \mathbb{S}_C = \pi * r^2 $$ $$ \mathbb{S}_N = (2*r)^2 = 4r^2 $$ Do đó, công thức trên trở thành: $$ \frac{N_C}{N} \approx \frac{\pi * r^2}{4r^2} = \frac{\pi}{4} $$ Suy ra, ta có giá trị số $\pi$ được tính như sau: $$ \pi \approx \frac{4N_C}{N} $$

Các bạn hãy lập trình tính sấp xỉ số $\pi$ dùng phương pháp Monte Carlo. Cho biết các sai số với $N = 100$, $N = 10000$, và $N = 1000000$. Vẽ đồ thị minh họa cho chương trình.

Hướng dẫn: Viết chương trình phát sinh $N$ điểm $(x_i, y_i)$ với $x_i, y_i \in [-1, -1]$. $(x_i, y_i)$ thuộc hình tròn tâm $(0, 0)$ bán kính 1 nội tiếp hình vuông có tâm tại $(0, 0)$ độ dài cạnh 2 khi và chỉ khi $x_i^2 + y_i^2 <= 1$.

import numpy as np

def calc_pi_monte_carlo(n=100):
    # Sinh n điểm ngẫu nhiên trong hình vuông [0,1)x[0,1)
    x = np.random.rand(n)
    y = np.random.rand(n)
    
    # Kiểm tra xem điểm có nằm trong hình tròn bán kính 1 không
    inside_circle = (x**2 + y**2) <= 1
    
    # Tỷ lệ điểm nằm trong hình tròn nhân 4 sẽ xấp xỉ pi
    pi = 4 * np.sum(inside_circle) / n
    return pi
